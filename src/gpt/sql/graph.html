async function downloadXlsx() { const url = `/api/genericos?${new URLSearchParams({ generics: true, score: true, xlsx: 'generate' })}`; const res = await fetch(url, { credentials: 'include' }); if (!res.ok) throw new Error('No se pudo generar el XLSX'); const blob = await res.blob(); // Intenta extraer filename del header const cd = res.headers.get('Content-Disposition') || ''; const m = cd.match(/filename\*?=(?:UTF-8'')?["']?([^"';\n]+)["']?/i); const filename = m?.[1] || `score_${new Date().toISOString().slice(0,10)}.xlsx`; const a = document.createElement('a'); a.href = URL.createObjectURL(blob); a.download = filename; document.body.appendChild(a); a.click(); a.remove(); setTimeout(() => URL.revokeObjectURL(a.href), 2000); } // Ejemplo: bot√≥n byId('btnXlsx')?.addEventListener('click', downloadXlsx);
